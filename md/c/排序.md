void insert_sort(int arr[], int len);
int main()
{
    int arr[]={9,3,4,2,6,7,5,1};
    int len = sizeof( arr ) / sizeof(int);

    insert_sort(arr,len);

    for (int i = 0; i < len; i++)
        printf("%d ", arr[i]);
    printf("\n");

    return 0;
}
void insert_sort(int arr[] , int len)
{
    for (int i = 0; i < len; i++)
    {
        int temp = arr[i];
        for (int j = i - 1; j >= 0; j--)
        {
            if( temp < arr[j] )
            {
                arr[j+1] = arr[j];
                arr[j] = temp; // 这句可以优化, 只在最后移动的那次赋值就 ok
            }
            else
            {
                break;
            }
        }
    }
}

// 优化版 1
void insert_sort2(int arr[] , int len)
{
    for (int i = 0, j = 0; i < len; i++)
    {
        int temp = arr[i];
        for (j = i - 1; j >= 0; j--)
        {
            if( arr[j] > temp )
                arr[j+1] = arr[j];
            else
                break;
        }
        arr[j+1] = temp;
    }
}

// 优化版 2
void insert_sort3(int arr[] , int len)
{
    for (int i = 0, j = 0; i < len; i++)
    {
        int temp = arr[i]; // 需要移动的项
        j = i - 1;         // 下一次要比较的下标
        while ( j >= 0 && arr[j] > temp )
        {
            arr[j+1] = arr[j]; // 将小于移动项的 向后移动一位
            j--;               // 保存的下一次比较的下标
        }
        arr[j+1] = temp;       // j+1 就是空出来的那个位置,将移动的项存进去就ok
    }
}
